class Promo {

  constructor() {
    this._name = "";
    this._code = "";
    this._rules = [];
  }

  set code(value) {
    this._code = value;
  }

  set name(value) {
    this._name = value;
  }

  addRule(rule) {
    this._rules.push(rule);
  }

  toString() {
    return `${this._code} ${this._name} rules: ${this._rules.length}`;
  }
}

const p = new Promo();
